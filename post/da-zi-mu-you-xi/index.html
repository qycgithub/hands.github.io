打字母游戏

· 2019-12-11 · # 经验项目

在一个300*400的窗体上，有随机产生的字母向下落。在键盘上敲字母，如果对了就消掉，初始成绩为1000分，每敲对一个字母加10分，如果字母落到屏幕下方，或者敲错，扣100分。

import java.awt.Color;
import java.awt.Graphics;
import java.awt.Panel;
import java.awt.event.KeyEvent;
import java.awt.event.KeyListener;

​
​ public class MyPanel extends Panel implements Runnable, KeyListener {
​

	int x[]=new int[10];
	int y[]=new int[10];
	char c[]=new char[10];
	int score=1000;
	MyPanel()
	{
		for(int i=0;i<10;i++)
		{
			x[i]=(int)(Math.random()*300);
			y[i]=(int)(Math.random()*300);
			c[i]=(char)(Math.random()*26+97);
		}
		
	}
	
	public void paint(Graphics g)
	{
		for(int i=0;i<10;i++)
		{
			g.drawString(new Character(c[i]).toString(), x[i], y[i]);
		}
		g.setColor(Color.RED);
		g.drawString("你的成绩是:"+score,5,15);
		
	}
	
	public void run()
	{
		while(true)
		{
			for(int i=0;i<10;i++)
			{
				y[i]++;
				if(y[i]>400)
				{
					y[i]=0;
					x[i]=(int)(Math.random()*300);
					c[i]=(char)(Math.random()*26+97);
					score-=100;//字母掉到屏幕外扣分
				}
			}
			try {
				Thread.sleep(30);
			}catch(Exception e) {}
			repaint();
			
		}
	}
